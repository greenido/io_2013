  url: apiUrl,
  dataType: 'json',
  contentType: 'application/json',
  type: "GET",
  success: function(data) {
    showList(data);
  },
  error: function(xhr, ajaxOptions, thrownError) {
    console.error("Picturesque list error: " + xhr.status);
  }   
});
